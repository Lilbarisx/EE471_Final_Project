\section{Impact \& Expected Outcomes}

\begin{frame}{Expected Impact}
    \begin{itemize}
        \item \textbf{Empowerment:} Enables elderly and visually impaired users to safely select and consume food or medicines independently.
        \item \textbf{Zero Latency AI:} Proves that heavy ML tasks (OCR, LLM, Whisper) can run efficiently on the Edge (local GPU) without requiring expensive Cloud GPUs.
        \item \textbf{Production-Ready DevOps:} Demonstrates full containerization (Docker) and professional cloud practices (AWS EC2/S3, CI/CD pipeline).
    \end{itemize}
\end{frame}

\begin{frame}{Demo Walkthrough: Safe Chocolate Scan}
    \begin{block}{1. User Health Profile}
        User Barış has a configured allergy profile on AWS: \textbf{Peanut (Yer Fıstığı) Allergy}.
    \end{block}
    \begin{block}{2. Scan and Local Inference}
        Barış takes a photo of a chocolate bar. Local OCR detects: \textit{"...Şeker, Kakao Yağı, Yer Fıstığı (\%12)..."}. SmolLM2 cross-references and flags the danger.
    \end{block}
    \begin{block}{3. Multi-Sensory Alert}
        Flutter app screen turns Red. Phone speaker speaks: \textit{"Dikkat Barış! Alerjiniz olan yer fıstığı tespit edildi. Lütfen tüketmeyin."}
    \end{block}
\end{frame}
