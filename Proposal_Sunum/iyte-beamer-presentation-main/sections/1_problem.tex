\section{Problem \& Motivation}

\begin{frame}{The Problem: Packaging Accessibility}
    \begin{itemize}
        \item \textbf{Text-Heavy Packaging:} Food ingredients and medication instructions are printed in very small fonts, making them inaccessible to visually impaired and elderly users.
        \item \textbf{Hidden Hazards:} Important allergen warnings (e.g., peanuts, gluten, dairy) or specific contraindications are often buried in dense text.
        \item \textbf{Language Barriers:} Travelers and immigrants face difficulty reading product packages in foreign languages, risking accidental exposure to critical allergens.
    \end{itemize}
\end{frame}

\begin{frame}{The Safety Risks}
    \begin{block}{Allergen Exposures}
        Accidental ingestion of allergens like peanuts or gluten can lead to severe health crises (e.g., anaphylaxis or severe celiac reactions).
    \end{block}
    \begin{block}{Medication Mismanagement}
        Misreading active ingredients or dosages poses high risks of drug-drug or food-drug interaction conflicts.
    \end{block}
\end{frame}
